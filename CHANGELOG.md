inputCombKey(metaKey, keyCode) // 组合键
inputCombKey(metaKey[], keyCode) // 多meta组合键

// meta 键对照:
// SHIFT == 1
// SHIFT_LEFT == 64
// SHIFT_RIGHT == 128
// CTRL == 4096
// CTRL_LEFT == 8192
// CTRL_RIGHT == 16384
// ALT == 2
// ALT_LEFT == 16
// ALT_RIGHT == 32

// 输入组合键: ctrl + shift + v:
